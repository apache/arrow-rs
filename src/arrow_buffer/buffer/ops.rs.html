22/// Apply a bitwise operation `op` to four inputs and return the result as a Buffer.
23/// The inputs are treated as bitmaps, meaning that offsets and length are specified in number of bits.
24pub fn bitwise_quaternary_op_helper<F>(
25    buffers: [&Buffer; 4],
26    offsets: [usize; 4],
27    len_in_bits: usize,
28    op: F,
29) -> Buffer
30where
31    F: Fn(u64, u64, u64, u64) -> u64,
32{
33    let first_chunks = buffers[0].bit_chunks(offsets[0], len_in_bits);
34    let second_chunks = buffers[1].bit_chunks(offsets[1], len_in_bits);
35    let third_chunks = buffers[2].bit_chunks(offsets[2], len_in_bits);
36    let fourth_chunks = buffers[3].bit_chunks(offsets[3], len_in_bits);
37
38    let chunks = first_chunks
39        .iter()
40        .zip(second_chunks.iter())
41        .zip(third_chunks.iter())
42        .zip(fourth_chunks.iter())
43        .map(|(((first, second), third), fourth)| op(first, second, third, fourth));
44    // Soundness: `BitChunks` is a `BitChunks` iterator which
45    // correctly reports its upper bound
46    let mut buffer = unsafe { MutableBuffer::from_trusted_len_iter(chunks) };
47
48    let remainder_bytes = ceil(first_chunks.remainder_len(), 8);
49    let rem = op(
50        first_chunks.remainder_bits(),
51        second_chunks.remainder_bits(),
52        third_chunks.remainder_bits(),
53        fourth_chunks.remainder_bits(),
54    );
55    // we are counting its starting from the least significant bit, to to_le_bytes should be correct
56    let rem = &rem.to_le_bytes()[0..remainder_bytes];
57    buffer.extend_from_slice(rem);
58
59    buffer.into()
60}

62/// Apply a bitwise operation `op` to two inputs and return the result as a Buffer.
63/// The inputs are treated as bitmaps, meaning that offsets and length are specified in number of bits.
64pub fn bitwise_bin_op_helper<F>(
65    left: &Buffer,
66    left_offset_in_bits: usize,
67    right: &Buffer,
68    right_offset_in_bits: usize,
69    len_in_bits: usize,
70    mut op: F,
71) -> Buffer
72where
73    F: FnMut(u64, u64) -> u64,
74{
75    let left_chunks = left.bit_chunks(left_offset_in_bits, len_in_bits);
76    let right_chunks = right.bit_chunks(right_offset_in_bits, len_in_bits);
77
78    let chunks = left_chunks
79        .iter()
80        .zip(right_chunks.iter())
81        .map(|(left, right)| op(left, right));
82    // Soundness: `BitChunks` is a `BitChunks` iterator which
83    // correctly reports its upper bound
84    let mut buffer = unsafe { MutableBuffer::from_trusted_len_iter(chunks) };
85
86    let remainder_bytes = ceil(left_chunks.remainder_len(), 8);
87    let rem = op(left_chunks.remainder_bits(), right_chunks.remainder_bits());
88    // we are counting its starting from the least significant bit, to to_le_bytes should be correct
89    let rem = &rem.to_le_bytes()[0..remainder_bytes];
90    buffer.extend_from_slice(rem);
91
92    buffer.into()
93}

95/// Apply a bitwise operation `op` to one input and return the result as a Buffer.
96/// The input is treated as a bitmap, meaning that offset and length are specified in number of bits.
97#[deprecated(
98    since = "57.2.0",
99    note = "use BooleanBuffer::from_bitwise_unary_op instead"
100)]
101pub fn bitwise_unary_op_helper<F>(
102    left: &Buffer,
103    offset_in_bits: usize,
104    len_in_bits: usize,
105    op: F,
106) -> Buffer
107where
108    F: FnMut(u64) -> u64,
109{
110    BooleanBuffer::from_bitwise_unary_op(left, offset_in_bits, len_in_bits, op).into_inner()
111}
