19use crate::schema::{AvroSchema, SCHEMA_METADATA_KEY};
20use crate::writer::encoder::write_long;
21use arrow_schema::{ArrowError, Schema};
22use rand::RngCore;
23use std::fmt::Debug;
24use std::io::Write;

26/// Format abstraction implemented by each container‐level writer.
27pub trait AvroFormat: Debug + Default {
28    /// If `true`, the writer for this format will query `single_object_prefix()`
29    /// and write the prefix before each record. If `false`, the writer can
30    /// skip this step. This is a performance hint for the writer.
31    const NEEDS_PREFIX: bool;
32
33    /// Write any bytes required at the very beginning of the output stream
34    /// (file header, etc.).
35    /// Implementations **must not** write any record data.
36    fn start_stream<W: Write>(
37        &mut self,
38        writer: &mut W,
39        schema: &Schema,
40        compression: Option<CompressionCodec>,
41    ) -> Result<(), ArrowError>;
42
43    /// Return the 16‑byte sync marker (OCF) or `None` (binary stream).
44    fn sync_marker(&self) -> Option<&[u8; 16]>;
45}
46
47/// Avro Object Container File (OCF) format writer.
48#[derive(Debug, Default)]
49pub struct AvroOcfFormat {
50    sync_marker: [u8; 16],
51}
52
53impl AvroFormat for AvroOcfFormat {
54    const NEEDS_PREFIX: bool = false;
55    fn start_stream<W: Write>(
56        &mut self,
57        writer: &mut W,
58        schema: &Schema,
59        compression: Option<CompressionCodec>,
60    ) -> Result<(), ArrowError> {
61        let mut rng = rand::rng();
62        rng.fill_bytes(&mut self.sync_marker);
63        // Choose the Avro schema JSON that the file will advertise.
64        // If `schema.metadata[SCHEMA_METADATA_KEY]` exists, AvroSchema::try_from
65        // uses it verbatim; otherwise it is generated from the Arrow schema.
66        let avro_schema = AvroSchema::try_from(schema)?;
67        // Magic
68        writer
69            .write_all(b"Obj\x01")
70            .map_err(|e| ArrowError::IoError(format!("write OCF magic: {e}"), e))?;
71        // File metadata map: { "avro.schema": <json>, "avro.codec": <codec> }
72        let codec_str = match compression {
73            Some(CompressionCodec::Deflate) => "deflate",
74            Some(CompressionCodec::Snappy) => "snappy",
75            Some(CompressionCodec::ZStandard) => "zstandard",
76            Some(CompressionCodec::Bzip2) => "bzip2",
77            Some(CompressionCodec::Xz) => "xz",
78            None => "null",
79        };
80        // Map block: count=2, then key/value pairs, then terminating count=0
81        write_long(writer, 2)?;
82        write_string(writer, SCHEMA_METADATA_KEY)?;
83        write_bytes(writer, avro_schema.json_string.as_bytes())?;
84        write_string(writer, CODEC_METADATA_KEY)?;
85        write_bytes(writer, codec_str.as_bytes())?;
86        write_long(writer, 0)?;
87        // Sync marker (16 bytes)
88        writer
89            .write_all(&self.sync_marker)
90            .map_err(|e| ArrowError::IoError(format!("write OCF sync marker: {e}"), e))?;
91        Ok(())
92    }
93
94    fn sync_marker(&self) -> Option<&[u8; 16]> {
95        Some(&self.sync_marker)
96    }
97}

99/// Raw Avro binary streaming format using **Single-Object Encoding** per record.
100///
101/// Each record written by the stream writer is framed with a prefix determined
102/// by the schema fingerprinting algorithm.
103///
104/// See: <https://avro.apache.org/docs/1.11.1/specification/#single-object-encoding>
105/// See: <https://docs.confluent.io/platform/current/schema-registry/fundamentals/serdes-develop/index.html#wire-format>
106#[derive(Debug, Default)]
107pub struct AvroBinaryFormat {}
108
109impl AvroFormat for AvroBinaryFormat {
110    const NEEDS_PREFIX: bool = true;
111    fn start_stream<W: Write>(
112        &mut self,
113        _writer: &mut W,
114        _schema: &Schema,
115        compression: Option<CompressionCodec>,
116    ) -> Result<(), ArrowError> {
117        if compression.is_some() {
118            return Err(ArrowError::InvalidArgumentError(
119                "Compression not supported for Avro binary streaming".to_string(),
120            ));
121        }
122
123        Ok(())
124    }
125
126    fn sync_marker(&self) -> Option<&[u8; 16]> {
127        None
128    }
129}
130
131#[inline]
132fn write_string<W: Write>(writer: &mut W, s: &str) -> Result<(), ArrowError> {
133    write_bytes(writer, s.as_bytes())
134}
135
136#[inline]
137fn write_bytes<W: Write>(writer: &mut W, bytes: &[u8]) -> Result<(), ArrowError> {
138    write_long(writer, bytes.len() as i64)?;
139    writer
140        .write_all(bytes)
141        .map_err(|e| ArrowError::IoError(format!("write bytes: {e}"), e))
142}
