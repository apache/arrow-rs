18use crate::errors::ParquetError;
19use crate::file::reader::{ChunkReader, Length};
20use bytes::Bytes;
21use std::fmt::Display;
22use std::ops::Range;
23
24/// Holds multiple buffers of data
25///
26/// This is the in-memory buffer for the ParquetDecoder and ParquetMetadataDecoders
27///
28/// Features:
29/// 1. Zero copy
30/// 2. non contiguous ranges of bytes
31///
32/// # Non Coalescing
33///
34/// This buffer does not coalesce  (merging adjacent ranges of bytes into a
35/// single range). Coalescing at this level would require copying the data but
36/// the caller may already have the needed data in a single buffer which would
37/// require no copying.
38///
39/// Thus, the implementation defers to the caller to coalesce subsequent requests
40/// if desired.
41#[derive(Debug, Clone)]
42pub(crate) struct PushBuffers {
43    /// the virtual "offset" of this buffers (added to any request)
44    offset: u64,
45    /// The total length of the file being decoded
46    file_len: u64,
47    /// The ranges of data that are available for decoding (not adjusted for offset)
48    ranges: Vec<Range<u64>>,
49    /// The buffers of data that can be used to decode the Parquet file
50    buffers: Vec<Bytes>,
51}
52
53impl Display for PushBuffers {
54    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
55        writeln!(
56            f,
57            "Buffers (offset: {}, file_len: {})",
58            self.offset, self.file_len
59        )?;
60        writeln!(f, "Available Ranges (w/ offset):")?;
61        for range in &self.ranges {
62            writeln!(
63                f,
64                "  {}..{} ({}..{}): {} bytes",
65                range.start,
66                range.end,
67                range.start + self.offset,
68                range.end + self.offset,
69                range.end - range.start
70            )?;
71        }
72
73        Ok(())
74    }
75}
76
77impl PushBuffers {
78    /// Create a new Buffers instance with the given file length
79    pub fn new(file_len: u64) -> Self {
80        Self {
81            offset: 0,
82            file_len,
83            ranges: Vec::new(),
84            buffers: Vec::new(),
85        }
86    }
87
88    /// Push all the ranges and buffers
89    pub fn push_ranges(&mut self, ranges: Vec<Range<u64>>, buffers: Vec<Bytes>) {
90        assert_eq!(
91            ranges.len(),
92            buffers.len(),
93            "Number of ranges must match number of buffers"
94        );
95        for (range, buffer) in ranges.into_iter().zip(buffers.into_iter()) {
96            self.push_range(range, buffer);
97        }
98    }
99
100    /// Push a new range and its associated buffer
101    pub fn push_range(&mut self, range: Range<u64>, buffer: Bytes) {
102        assert_eq!(
103            (range.end - range.start) as usize,
104            buffer.len(),
105            "Range length must match buffer length"
106        );
107        self.ranges.push(range);
108        self.buffers.push(buffer);
109    }
110
111    /// Returns true if the Buffers contains data for the given range
112    pub fn has_range(&self, range: &Range<u64>) -> bool {
113        self.ranges
114            .iter()
115            .any(|r| r.start <= range.start && r.end >= range.end)
116    }
117
118    fn iter(&self) -> impl Iterator<Item = (&Range<u64>, &Bytes)> {
119        self.ranges.iter().zip(self.buffers.iter())
120    }
121
122    /// return the file length of the Parquet file being read
123    pub fn file_len(&self) -> u64 {
124        self.file_len
125    }
126
127    /// Specify a new offset
128    pub fn with_offset(mut self, offset: u64) -> Self {
129        self.offset = offset;
130        self
131    }
132
133    /// Return the total of all buffered ranges
134    #[cfg(feature = "arrow")]
135    pub fn buffered_bytes(&self) -> u64 {
136        self.ranges.iter().map(|r| r.end - r.start).sum()
137    }
138
139    /// Clear any range and corresponding buffer that is exactly in the ranges_to_clear
140    #[cfg(feature = "arrow")]
141    pub fn clear_ranges(&mut self, ranges_to_clear: &[Range<u64>]) {
142        let mut new_ranges = Vec::new();
143        let mut new_buffers = Vec::new();
144
145        for (range, buffer) in self.iter() {
146            if !ranges_to_clear
147                .iter()
148                .any(|r| r.start == range.start && r.end == range.end)
149            {
150                new_ranges.push(range.clone());
151                new_buffers.push(buffer.clone());
152            }
153        }
154        self.ranges = new_ranges;
155        self.buffers = new_buffers;
156    }
157}
158
159impl Length for PushBuffers {
160    fn len(&self) -> u64 {
161        self.file_len
162    }
163}
164
165/// less efficient implementation of Read for Buffers
166impl std::io::Read for PushBuffers {
167    fn read(&mut self, buf: &mut [u8]) -> std::io::Result<usize> {
168        // Find the range that contains the start offset
169        let mut found = false;
170        for (range, data) in self.iter() {
171            if range.start <= self.offset && range.end >= self.offset + buf.len() as u64 {
172                // Found the range, figure out the starting offset in the buffer
173                let start_offset = (self.offset - range.start) as usize;
174                let end_offset = start_offset + buf.len();
175                let slice = data.slice(start_offset..end_offset);
176                buf.copy_from_slice(slice.as_ref());
177                found = true;
178                break;
179            }
180        }
181        if found {
182            // If we found the range, we can return the number of bytes read
183            // advance our offset
184            self.offset += buf.len() as u64;
185            Ok(buf.len())
186        } else {
187            Err(std::io::Error::new(
188                std::io::ErrorKind::UnexpectedEof,
189                "No data available in Buffers",
190            ))
191        }
192    }
193}
194
195impl ChunkReader for PushBuffers {
196    type T = Self;
197
198    fn get_read(&self, start: u64) -> Result<Self::T, ParquetError> {
199        Ok(self.clone().with_offset(self.offset + start))
200    }
201
202    fn get_bytes(&self, start: u64, length: usize) -> Result<Bytes, ParquetError> {
203        if start > self.file_len {
204            return Err(ParquetError::General(format!(
205                "Requested start {start} is beyond the end of the file (file length: {})",
206                self.file_len
207            )));
208        }
209
210        // find the range that contains the start offset
211        for (range, data) in self.iter() {
212            if range.start <= start && range.end >= start + length as u64 {
213                // Found the range, figure out the starting offset in the buffer
214                let start_offset = (start - range.start) as usize;
215                return Ok(data.slice(start_offset..start_offset + length));
216            }
217        }
218        // Signal that we need more data
219        let requested_end = start + length as u64;
220        Err(ParquetError::NeedMoreDataRange(start..requested_end))
221    }
222}
