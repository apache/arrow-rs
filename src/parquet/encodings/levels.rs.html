25/// Encoder for definition/repetition levels.
26/// Currently only supports Rle and BitPacked (dev/null) encoding, including v2.
27pub enum LevelEncoder {
28    Rle(RleEncoder),
29    RleV2(RleEncoder),
30}
31
32impl LevelEncoder {
33    /// Creates a new streaming level encoder for Data Page v1.
34    ///
35    /// This does not require knowing the number of values
36    /// upfront, making it suitable for incremental encoding where levels are fed in
37    /// as they arrive via [`put_with_observer`](Self::put_with_observer).
38    pub fn v1_streaming(max_level: i16) -> Self {
39        let bit_width = num_required_bits(max_level as u64);
40        // Reserve space for length header
41        let buffer = vec![0u8; 4];
42        LevelEncoder::Rle(RleEncoder::new_from_buf(bit_width, buffer))
43    }
44
45    /// Creates a new streaming RLE level encoder for Data Page v2.
46    ///
47    /// This does not require knowing the number of values
48    /// upfront, making it suitable for incremental encoding where levels are fed in
49    /// as they arrive via [`put_with_observer`](Self::put_with_observer).
50    pub fn v2_streaming(max_level: i16) -> Self {
51        let bit_width = num_required_bits(max_level as u64);
52        LevelEncoder::RleV2(RleEncoder::new_from_buf(bit_width, Vec::new()))
53    }
54
55    /// Put/encode levels vector into this level encoder and call
56    /// `observer(value, count)` for each run of identical values encountered
57    /// during encoding.
58    ///
59    /// Returns number of encoded values that are less than or equal to length
60    /// of the input buffer.
61    ///
62    /// This method does **not** flush the underlying encoder, so it can be called
63    /// incrementally across multiple batches without forcing run boundaries.
64    /// The encoder is flushed automatically when [`consume`](Self::consume) is called.
65    #[inline]
66    pub fn put_with_observer<F>(&mut self, buffer: &[i16], mut observer: F) -> usize
67    where
68        F: FnMut(i16, usize),
69    {
70        match *self {
71            LevelEncoder::Rle(ref mut encoder) | LevelEncoder::RleV2(ref mut encoder) => {
72                let mut remaining = buffer;
73                while let Some((&value, rest)) = remaining.split_first() {
74                    encoder.put(value as u64);
75                    // After put(), check if the encoder just entered RLE
76                    // accumulation mode. If so, scan ahead for the rest of
77                    // this run to batch the observer call and bulk-extend.
78                    if encoder.is_accumulating_rle(value as u64) {
79                        let run_len = rest.iter().take_while(|&&v| v == value).count();
80                        if run_len > 0 {
81                            encoder.extend_run(run_len);
82                        }
83                        observer(value, 1 + run_len);
84                        remaining = &rest[run_len..];
85                    } else {
86                        observer(value, 1);
87                        remaining = rest;
88                    }
89                }
90                buffer.len()
91            }
92        }
93    }
94
95    /// Encode `count` repetitions of a single level value, calling
96    /// `observer(value, count)` exactly once.
97    ///
98    /// This can be used to encode uniform runs without allocating a level
99    /// buffer. Like [`Self::put_with_observer`], but specialized for a single
100    /// repeated level value.
101    ///
102    /// This is O(1) amortized for RLE-based encoders (after a small warmup).
103    #[inline]
104    pub fn put_n_with_observer<F>(&mut self, value: i16, count: usize, mut observer: F)
105    where
106        F: FnMut(i16, usize),
107    {
108        let encoder = match *self {
109            LevelEncoder::Rle(ref mut encoder) | LevelEncoder::RleV2(ref mut encoder) => encoder,
110        };
111
112        // Feed values individually until the encoder enters RLE accumulation
113        // mode for this value, or until we've encoded everything.
114        let mut remaining = count;
115        while remaining > 0 && !encoder.is_accumulating_rle(value as u64) {
116            encoder.put(value as u64);
117            remaining -= 1;
118        }
119        // If we're now in accumulation mode, bulk-extend the rest.
120        if remaining > 0 {
121            encoder.extend_run(remaining);
122        }
123        observer(value, count);
124    }
125
126    /// Finalizes level encoder, flush all intermediate buffers and return resulting
127    /// encoded buffer. Returned buffer is already truncated to encoded bytes only.
128    #[inline]
129    #[allow(unused)]
130    pub fn consume(self) -> Vec<u8> {
131        match self {
132            LevelEncoder::Rle(encoder) => {
133                let mut encoded_data = encoder.consume();
134                // Account for the buffer offset
135                let encoded_len = encoded_data.len() - mem::size_of::<i32>();
136                let len = (encoded_len as i32).to_le();
137                let len_bytes = len.as_bytes();
138                encoded_data[0..len_bytes.len()].copy_from_slice(len_bytes);
139                encoded_data
140            }
141            LevelEncoder::RleV2(encoder) => encoder.consume(),
142        }
143    }
144
145    /// Flushes all intermediate buffers, passes the encoded data to `f`, then
146    /// resets the encoder for reuse while retaining the buffer allocation.
147    #[inline]
148    pub fn flush_to<F, R>(&mut self, f: F) -> R
149    where
150        F: FnOnce(&[u8]) -> R,
151    {
152        let result = match self {
153            LevelEncoder::Rle(encoder) => {
154                let data = encoder.flush_buffer_mut();
155                // Patch the 4-byte length header reserved at the start of the buffer
156                let encoded_len = (data.len() - mem::size_of::<i32>()) as i32;
157                data[..4].copy_from_slice(&encoded_len.to_le_bytes());
158                f(data)
159            }
160            LevelEncoder::RleV2(encoder) => f(encoder.flush_buffer()),
161        };
162        match self {
163            LevelEncoder::Rle(encoder) => {
164                encoder.clear();
165                // Re-reserve the 4-byte length header for the next page
166                encoder.skip(mem::size_of::<i32>());
167            }
168            LevelEncoder::RleV2(encoder) => encoder.clear(),
169        }
170        result
171    }
172}

174#[cfg(test)]
175mod tests {
176    use super::*;
177
178    /// Encode `count` repetitions of `value` using `put_with_observer` and
179    /// return the raw encoded bytes.
180    fn reference_encode(max_level: i16, values: &[i16]) -> Vec<u8> {
181        let mut enc = LevelEncoder::v2_streaming(max_level);
182        enc.put_with_observer(values, |_, _| {});
183        enc.consume()
184    }
185
186    #[test]
187    fn test_put_n_with_observer_large_run() {
188        // Large count exercises the bulk extend_run path (past the 8-value warmup).
189        let max_level = 3;
190        let count = 10_000;
191        let mut enc = LevelEncoder::v2_streaming(max_level);
192        enc.put_n_with_observer(2, count, |_, _| {});
193        assert_eq!(enc.consume(), reference_encode(max_level, &vec![2; count]));
194    }
195
196    #[test]
197    fn test_put_n_with_observer_small_count() {
198        // Count smaller than the RLE warmup threshold — only the per-element loop runs.
199        let max_level = 3;
200        let mut enc = LevelEncoder::v2_streaming(max_level);
201        enc.put_n_with_observer(1, 5, |_, _| {});
202        assert_eq!(enc.consume(), reference_encode(max_level, &[1; 5]));
203    }
204
205    #[test]
206    fn test_put_n_with_observer_exact_threshold() {
207        // Exactly 8 values: the warmup loop completes and extend_run gets 0.
208        let max_level = 3;
209        let mut enc = LevelEncoder::v2_streaming(max_level);
210        enc.put_n_with_observer(3, 8, |_, _| {});
211        assert_eq!(enc.consume(), reference_encode(max_level, &[3; 8]));
212    }

246    #[test]
247    fn test_put_n_with_observer_followed_by_different_value() {
248        // Two consecutive put_n calls with different values — verifies that
249        // the encoder correctly transitions between runs.
250        let max_level = 3;
251        let mut enc = LevelEncoder::v2_streaming(max_level);
252        let mut calls: Vec<(i16, usize)> = Vec::new();
253        enc.put_n_with_observer(1, 100, |v, c| calls.push((v, c)));
254        enc.put_n_with_observer(3, 200, |v, c| calls.push((v, c)));
255        assert_eq!(calls, vec![(1, 100), (3, 200)]);
256
257        let reference = reference_encode(max_level, &[&[1i16; 100][..], &[3i16; 200]].concat());
258        assert_eq!(enc.consume(), reference);
259    }

261    #[test]
262    fn test_put_n_with_observer_interleaved_with_put_with_observer() {
263        // Mix put_n_with_observer and put_with_observer to verify they compose.
264        let max_level = 3;
265        let mut enc = LevelEncoder::v2_streaming(max_level);
266        enc.put_n_with_observer(2, 50, |_, _| {});
267        enc.put_with_observer(&[0, 0, 1, 1, 3], |_, _| {});
268        enc.put_n_with_observer(2, 50, |_, _| {});
269
270        let input = [&[2i16; 50][..], &[0, 0, 1, 1, 3], &[2i16; 50]].concat();
271        assert_eq!(enc.consume(), reference_encode(max_level, &input));
272    }
