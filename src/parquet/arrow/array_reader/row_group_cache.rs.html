18use arrow_array::{Array, ArrayRef};
19use arrow_schema::DataType;
20use std::collections::HashMap;
21
22/// Starting row ID for this batch
23///
24/// The `BatchID` is used to identify batches of rows within a row group.
25///
26/// The row_index in the id are relative to the rows being read from the
27/// underlying column reader (which might already have a RowSelection applied)
28///
29/// The `BatchID` for any particular row is `row_index / batch_size`. The
30/// integer division ensures that rows in the same batch share the same
31/// the BatchID which can be calculated quickly from the row index
32#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
33pub struct BatchID {
34    pub val: usize,
35}
36
37/// Cache key that uniquely identifies a batch within a row group
38#[derive(Debug, Clone, PartialEq, Eq, Hash)]
39pub struct CacheKey {
40    /// Column index in the row group
41    pub column_idx: usize,
42    /// Starting row ID for this batch
43    pub batch_id: BatchID,
44}
45
46fn get_array_memory_size_for_cache(array: &ArrayRef) -> usize {
47    match array.data_type() {
48        // TODO: this is temporary workaround. It's very difficult to measure the actual memory usage of one StringViewArray,
49        // because the underlying buffer is shared with multiple StringViewArrays.
50        DataType::Utf8View => {
51            use arrow_array::cast::AsArray;
52            let array = array.as_string_view();
53            array.len() * 16 + array.total_buffer_bytes_used() + std::mem::size_of_val(array)
54        }
55        _ => array.get_array_memory_size(),
56    }
57}

59/// Row group cache that stores decoded arrow arrays at batch granularity
60///
61/// This cache is designed to avoid duplicate decoding when the same column
62/// appears in both filter predicates and output projection.
63#[derive(Debug)]
64pub struct RowGroupCache {
65    /// Cache storage mapping (column_idx, row_id) -> ArrayRef
66    cache: HashMap<CacheKey, ArrayRef>,
67    /// Cache granularity
68    batch_size: usize,
69    /// Maximum cache size in bytes
70    max_cache_bytes: usize,
71    /// Current cache size in bytes
72    current_cache_size: usize,
73}
74
75impl RowGroupCache {
76    /// Creates a new empty row group cache
77    pub fn new(batch_size: usize, max_cache_bytes: usize) -> Self {
78        Self {
79            cache: HashMap::new(),
80            batch_size,
81            max_cache_bytes,
82            current_cache_size: 0,
83        }
84    }
85
86    /// Inserts an array into the cache for the given column and starting row ID
87    /// Returns true if the array was inserted, false if it would exceed the cache size limit
88    pub fn insert(&mut self, column_idx: usize, batch_id: BatchID, array: ArrayRef) -> bool {
89        let array_size = get_array_memory_size_for_cache(&array);
90
91        // Check if adding this array would exceed the cache size limit
92        if self.current_cache_size + array_size > self.max_cache_bytes {
93            return false; // Cache is full, don't insert
94        }
95
96        let key = CacheKey {
97            column_idx,
98            batch_id,
99        };
100
101        let existing = self.cache.insert(key, array);
102        assert!(existing.is_none());
103        self.current_cache_size += array_size;
104        true
105    }
106
107    /// Retrieves a cached array for the given column and row ID
108    /// Returns None if not found in cache
109    pub fn get(&self, column_idx: usize, batch_id: BatchID) -> Option<ArrayRef> {
110        let key = CacheKey {
111            column_idx,
112            batch_id,
113        };
114        self.cache.get(&key).cloned()
115    }
116
117    /// Gets the batch size for this cache
118    pub fn batch_size(&self) -> usize {
119        self.batch_size
120    }
121
122    /// Removes a cached array for the given column and row ID
123    /// Returns true if the entry was found and removed, false otherwise
124    pub fn remove(&mut self, column_idx: usize, batch_id: BatchID) -> bool {
125        let key = CacheKey {
126            column_idx,
127            batch_id,
128        };
129        if let Some(array) = self.cache.remove(&key) {
130            self.current_cache_size -= get_array_memory_size_for_cache(&array);
131            true
132        } else {
133            false
134        }
135    }
136}

138#[cfg(test)]
139mod tests {
140    use super::*;
141    use arrow_array::{ArrayRef, Int32Array};
142    use std::sync::Arc;
143
144    #[test]
145    fn test_cache_basic_operations() {
146        let mut cache = RowGroupCache::new(1000, usize::MAX);
147
148        // Create test array
149        let array: ArrayRef = Arc::new(Int32Array::from(vec![1, 2, 3, 4, 5]));
150
151        // Test insert and get
152        let batch_id = BatchID { val: 0 };
153        assert!(cache.insert(0, batch_id, array.clone()));
154        let retrieved = cache.get(0, batch_id);
155        assert!(retrieved.is_some());
156        assert_eq!(retrieved.unwrap().len(), 5);
157
158        // Test miss
159        let miss = cache.get(1, batch_id);
160        assert!(miss.is_none());
161
162        // Test different row_id
163        let miss = cache.get(0, BatchID { val: 1000 });
164        assert!(miss.is_none());
165    }
166
167    #[test]
168    fn test_cache_remove() {
169        let mut cache = RowGroupCache::new(1000, usize::MAX);
170
171        // Create test arrays
172        let array1: ArrayRef = Arc::new(Int32Array::from(vec![1, 2, 3]));
173        let array2: ArrayRef = Arc::new(Int32Array::from(vec![4, 5, 6]));
174
175        // Insert arrays
176        assert!(cache.insert(0, BatchID { val: 0 }, array1.clone()));
177        assert!(cache.insert(0, BatchID { val: 1000 }, array2.clone()));
178        assert!(cache.insert(1, BatchID { val: 0 }, array1.clone()));
179
180        // Verify they're there
181        assert!(cache.get(0, BatchID { val: 0 }).is_some());
182        assert!(cache.get(0, BatchID { val: 1000 }).is_some());
183        assert!(cache.get(1, BatchID { val: 0 }).is_some());
184
185        // Remove one entry
186        let removed = cache.remove(0, BatchID { val: 0 });
187        assert!(removed);
188        assert!(cache.get(0, BatchID { val: 0 }).is_none());
189
190        // Other entries should still be there
191        assert!(cache.get(0, BatchID { val: 1000 }).is_some());
192        assert!(cache.get(1, BatchID { val: 0 }).is_some());
193
194        // Try to remove non-existent entry
195        let not_removed = cache.remove(0, BatchID { val: 0 });
196        assert!(!not_removed);
197
198        // Remove remaining entries
199        assert!(cache.remove(0, BatchID { val: 1000 }));
200        assert!(cache.remove(1, BatchID { val: 0 }));
201
202        // Cache should be empty
203        assert!(cache.get(0, BatchID { val: 1000 }).is_none());
204        assert!(cache.get(1, BatchID { val: 0 }).is_none());
205    }
206}
