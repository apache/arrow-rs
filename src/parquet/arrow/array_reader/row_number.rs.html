18use crate::arrow::array_reader::ArrayReader;
19use crate::errors::{ParquetError, Result};
20use crate::file::metadata::{ParquetMetaData, RowGroupMetaData};
21use arrow_array::{ArrayRef, Int64Array};
22use arrow_schema::DataType;
23use std::any::Any;
24use std::collections::HashMap;
25use std::sync::Arc;
26
27pub(crate) struct RowNumberReader {
28    buffered_row_numbers: Vec<i64>,
29    remaining_row_numbers: std::iter::Flatten<std::vec::IntoIter<std::ops::Range<i64>>>,
30}
31
32impl RowNumberReader {
33    pub(crate) fn try_new<'a>(
34        parquet_metadata: &'a ParquetMetaData,
35        row_groups: impl Iterator<Item = &'a RowGroupMetaData>,
36    ) -> Result<Self> {
37        // Pass 1: Build a map from ordinal to first_row_index
38        // This is O(M) where M is the total number of row groups in the file
39        let mut ordinal_to_offset: HashMap<i16, i64> = HashMap::new();
40        let mut first_row_index: i64 = 0;
41
42        for rg in parquet_metadata.row_groups() {
43            if let Some(ordinal) = rg.ordinal() {
44                ordinal_to_offset.insert(ordinal, first_row_index);
45            }
46            first_row_index += rg.num_rows();
47        }
48
49        // Pass 2: Build ranges in the order specified by the row_groups iterator
50        // This is O(N) where N is the number of selected row groups
51        // This preserves the user's requested order instead of sorting by ordinal
52        let ranges: Vec<_> = row_groups
53            .map(|rg| {
54                let ordinal = rg.ordinal().ok_or_else(|| {
55                    ParquetError::General(
56                        "Row group missing ordinal field, required to compute row numbers"
57                            .to_string(),
58                    )
59                })?;
60
61                let offset = ordinal_to_offset.get(&ordinal).ok_or_else(|| {
62                    ParquetError::General(format!(
63                        "Row group with ordinal {} not found in metadata",
64                        ordinal
65                    ))
66                })?;
67
68                Ok(*offset..*offset + rg.num_rows())
69            })
70            .collect::<Result<_>>()?;
71
72        Ok(Self {
73            buffered_row_numbers: Vec::new(),
74            remaining_row_numbers: ranges.into_iter().flatten(),
75        })
76    }
77}
78
79impl ArrayReader for RowNumberReader {
80    fn read_records(&mut self, batch_size: usize) -> Result<usize> {
81        let starting_len = self.buffered_row_numbers.len();
82        self.buffered_row_numbers
83            .extend((&mut self.remaining_row_numbers).take(batch_size));
84        Ok(self.buffered_row_numbers.len() - starting_len)
85    }
86
87    fn skip_records(&mut self, num_records: usize) -> Result<usize> {
88        // TODO: Use advance_by when it stabilizes to improve performance
89        Ok((&mut self.remaining_row_numbers).take(num_records).count())
90    }
91
92    fn as_any(&self) -> &dyn Any {
93        self
94    }
95
96    fn get_data_type(&self) -> &DataType {
97        &DataType::Int64
98    }
99
100    fn consume_batch(&mut self) -> Result<ArrayRef> {
101        Ok(Arc::new(Int64Array::from_iter(
102            self.buffered_row_numbers.drain(..),
103        )))
104    }
105
106    fn get_def_levels(&self) -> Option<&[i16]> {
107        None
108    }
109
110    fn get_rep_levels(&self) -> Option<&[i16]> {
111        None
112    }
113}

115#[cfg(test)]
116mod tests {
117    use super::*;
118    use crate::basic::Type as PhysicalType;
119    use crate::file::metadata::{
120        ColumnChunkMetaData, FileMetaData, ParquetMetaData, RowGroupMetaData,
121    };
122    use crate::schema::types::{SchemaDescriptor, Type as SchemaType};
123    use std::sync::Arc;
124
125    fn create_test_schema() -> Arc<SchemaDescriptor> {
126        let schema = SchemaType::group_type_builder("schema")
127            .with_fields(vec![Arc::new(
128                SchemaType::primitive_type_builder("test_col", PhysicalType::INT32)
129                    .build()
130                    .unwrap(),
131            )])
132            .build()
133            .unwrap();
134        Arc::new(SchemaDescriptor::new(Arc::new(schema)))
135    }
136
137    fn create_test_parquet_metadata(row_groups: Vec<(i16, i64)>) -> ParquetMetaData {
138        let schema_descr = create_test_schema();
139
140        let mut row_group_metas = vec![];
141        for (ordinal, num_rows) in row_groups {
142            let columns: Vec<_> = schema_descr
143                .columns()
144                .iter()
145                .map(|col| ColumnChunkMetaData::builder(col.clone()).build().unwrap())
146                .collect();
147
148            let row_group = RowGroupMetaData::builder(schema_descr.clone())
149                .set_num_rows(num_rows)
150                .set_ordinal(ordinal)
151                .set_total_byte_size(100)
152                .set_column_metadata(columns)
153                .build()
154                .unwrap();
155            row_group_metas.push(row_group);
156        }
157
158        let total_rows: i64 = row_group_metas.iter().map(|rg| rg.num_rows()).sum();
159        let file_metadata = FileMetaData::new(
160            1,            // version
161            total_rows,   // num_rows
162            None,         // created_by
163            None,         // key_value_metadata
164            schema_descr, // schema_descr
165            None,         // column_orders
166        );
167
168        ParquetMetaData::new(file_metadata, row_group_metas)
169    }
170
171    #[test]
172    fn test_row_number_reader_reverse_order() {
173        // Create metadata with 3 row groups, each with 2 rows
174        let metadata = create_test_parquet_metadata(vec![
175            (0, 2), // Row group 0: ordinal=0, rows 0-1
176            (1, 2), // Row group 1: ordinal=1, rows 2-3
177            (2, 2), // Row group 2: ordinal=2, rows 4-5
178        ]);
179
180        // Select only row groups with ordinals 2 and 0 (in that order)
181        // This means we want row group 2 first, then row group 0, skipping row group 1
182        let selected_row_groups: Vec<_> = vec![
183            &metadata.row_groups()[2], // ordinal 2
184            &metadata.row_groups()[0], // ordinal 0
185        ];
186
187        let mut reader =
188            RowNumberReader::try_new(&metadata, selected_row_groups.into_iter()).unwrap();
189
190        // Read all row numbers
191        let num_read = reader.read_records(6).unwrap();
192        assert_eq!(num_read, 4); // Should read 4 rows total (2 from each selected group)
193
194        let array = reader.consume_batch().unwrap();
195        let row_numbers = array.as_any().downcast_ref::<Int64Array>().unwrap();
196
197        // Expected: row group 2 first (rows 4-5), then row group 0 (rows 0-1)
198        let expected = vec![4, 5, 0, 1];
199        let actual: Vec<i64> = row_numbers.iter().map(|v| v.unwrap()).collect();
200
201        assert_eq!(
202            actual, expected,
203            "Row numbers should match the order of selected row groups, not file order"
204        );
205    }
