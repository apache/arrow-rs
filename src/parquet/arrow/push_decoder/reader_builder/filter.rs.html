18//! [`FilterInfo`] state machine for evaluating row filters
19
20use crate::arrow::ProjectionMask;
21use crate::arrow::array_reader::{CacheOptionsBuilder, RowGroupCache};
22use crate::arrow::arrow_reader::{ArrowPredicate, RowFilter};
23use std::num::NonZeroUsize;
24use std::sync::{Arc, Mutex};
25
26/// State machine for evaluating a sequence of predicates.
27///
28/// The `FilterInfo` owns the [`RowFilter`] being evaluated and tracks the current
29/// predicate to evaluate.
30#[derive(Debug)]
31pub(super) struct FilterInfo {
32    /// The predicates to evaluate, in order
33    ///
34    /// RowFilter is owned by `FilterInfo` because they may be mutated as part
35    /// of evaluation. Specifically, [`ArrowPredicate`] requires &mut self for
36    /// evaluation.
37    filter: RowFilter,
38    /// The next filter to be evaluated
39    next_predicate: NonZeroUsize,
40    /// Previously computed filter results
41    cache_info: CacheInfo,
42}
43
44/// Predicate cache
45///
46/// Note this is basically the same as CacheOptionsBuilder
47/// but it owns the ProjectionMask and RowGroupCache
48#[derive(Debug)]
49pub(super) struct CacheInfo {
50    /// The columns to cache in the predicate cache.
51    /// Normally these are the columns that filters may look at such that
52    /// if we have a filter like `(a + 10 > 5) AND (a + b = 0)` we cache `a` to avoid re-reading it between evaluating `a + 10 > 5` and `a + b = 0`.
53    cache_projection: ProjectionMask,
54    row_group_cache: Arc<Mutex<RowGroupCache>>,
55}
56
57impl CacheInfo {
58    pub(super) fn new(
59        cache_projection: ProjectionMask,
60        row_group_cache: Arc<Mutex<RowGroupCache>>,
61    ) -> Self {
62        Self {
63            cache_projection,
64            row_group_cache,
65        }
66    }
67
68    pub(super) fn builder(&self) -> CacheOptionsBuilder<'_> {
69        CacheOptionsBuilder::new(&self.cache_projection, &self.row_group_cache)
70    }
71}
72
73pub(super) enum AdvanceResult {
74    /// Advanced to the next predicate
75    Continue(FilterInfo),
76    /// No more predicates returns the row filter and cache info
77    Done(RowFilter, CacheInfo),
78}
79
80impl FilterInfo {
81    /// Create a new FilterInfo
82    pub(super) fn new(filter: RowFilter, cache_info: CacheInfo) -> Self {
83        Self {
84            filter,
85            next_predicate: NonZeroUsize::new(1).expect("1 is always non-zero"),
86            cache_info,
87        }
88    }
89
90    /// Advance to the next predicate
91    ///
92    /// Returns
93    /// * [`AdvanceResult::Continue`] returning the `FilterInfo` if there are
94    ///   more predicate to evaluate.
95    /// * [`AdvanceResult::Done`] with the inner [`RowFilter`] and [`CacheInfo]`
96    ///   if there are no more predicates
97    pub(super) fn advance(mut self) -> AdvanceResult {
98        if self.next_predicate.get() >= self.filter.predicates.len() {
99            AdvanceResult::Done(self.filter, self.cache_info)
100        } else {
101            self.next_predicate = self
102                .next_predicate
103                .checked_add(1)
104                .expect("no usize overflow");
105            AdvanceResult::Continue(self)
106        }
107    }
108
109    /// Return a mutable reference to the current predicate
110    pub(super) fn current_mut(&mut self) -> &mut dyn ArrowPredicate {
111        self.filter
112            .predicates
113            .get_mut(self.next_predicate.get() - 1)
114            // advance ensures next_predicate is always in bounds
115            .unwrap()
116            .as_mut()
117    }
118
119    /// Return the current predicate to evaluate
120    pub(super) fn current(&self) -> &dyn ArrowPredicate {
121        self.filter
122            .predicates
123            .get(self.next_predicate.get() - 1)
124            // advance ensures next_predicate is always in bounds
125            .unwrap()
126            .as_ref()
127    }
128
129    /// Return a reference to the cache projection
130    pub(super) fn cache_projection(&self) -> &ProjectionMask {
131        &self.cache_info.cache_projection
132    }
133
134    /// Return a cache builder to save the results of predicate evaluation
135    pub(super) fn cache_builder(&self) -> CacheOptionsBuilder<'_> {
136        self.cache_info.builder()
137    }
138
139    /// Returns the inner filter, consuming this FilterInfo
140    pub(super) fn into_filter(self) -> RowFilter {
141        self.filter
142    }
143}
