18//! Module for converting Variant data to JSON format

26// Format string constants to avoid duplication and reduce errors
27const DATE_FORMAT: &str = "%Y-%m-%d";
28const TIMESTAMP_NTZ_FORMAT: &str = "%Y-%m-%dT%H:%M:%S%.6f";
29
30// Helper functions for consistent formatting
31fn format_date_string(date: &chrono::NaiveDate) -> String {
32    date.format(DATE_FORMAT).to_string()
33}
34
35fn format_timestamp_ntz_string(ts: &chrono::NaiveDateTime) -> String {
36    ts.format(TIMESTAMP_NTZ_FORMAT).to_string()
37}
38
39fn format_binary_base64(bytes: &[u8]) -> String {
40    general_purpose::STANDARD.encode(bytes)
41}

43///
44/// This function writes JSON directly to any type that implements [`Write`],
45/// making it efficient for streaming or when you want to control the output destination.
46///
47/// See [`variant_to_json_string`] for a convenience function that returns a
48/// JSON string.
49///
50/// # Arguments
51///
52/// * `writer` - Writer to output JSON to
53/// * `variant` - The Variant value to convert
54///
55/// # Returns
56///
57/// * `Ok(())` if successful
58/// * `Err` with error details if conversion fails
59///
60/// # Examples
61///
62///
63/// ```rust
64/// # use parquet_variant::{Variant, variant_to_json};
65/// # use arrow_schema::ArrowError;
66/// let variant = Variant::from("Hello, World!");
67/// let mut buffer = Vec::new();
68/// variant_to_json(&mut buffer, &variant)?;
69/// assert_eq!(String::from_utf8(buffer).unwrap(), "\"Hello, World!\"");
70/// # Ok::<(), ArrowError>(())
71/// ```
72///
73/// # Example: Create a [`Variant::Object`] and convert to JSON
74/// ```rust
75/// # use parquet_variant::{Variant, VariantBuilder, variant_to_json};
76/// # use arrow_schema::ArrowError;
77/// let mut builder = VariantBuilder::new();
78/// // Create an object builder that will write fields to the object
79/// let mut object_builder = builder.new_object();
80/// object_builder.insert("first_name", "Jiaying");
81/// object_builder.insert("last_name", "Li");
82/// object_builder.finish();
83/// // Finish the builder to get the metadata and value
84/// let (metadata, value) = builder.finish();
85/// // Create the Variant and convert to JSON
86/// let variant = Variant::try_new(&metadata, &value)?;
87/// let mut writer = Vec::new();
88/// variant_to_json(&mut writer, &variant,)?;
89/// assert_eq!(br#"{"first_name":"Jiaying","last_name":"Li"}"#, writer.as_slice());
90/// # Ok::<(), ArrowError>(())
91/// ```
92pub fn variant_to_json(json_buffer: &mut impl Write, variant: &Variant) -> Result<(), ArrowError> {
93    match variant {
94        Variant::Null => write!(json_buffer, "null")?,
95        Variant::BooleanTrue => write!(json_buffer, "true")?,
96        Variant::BooleanFalse => write!(json_buffer, "false")?,
97        Variant::Int8(i) => write!(json_buffer, "{i}")?,
98        Variant::Int16(i) => write!(json_buffer, "{i}")?,
99        Variant::Int32(i) => write!(json_buffer, "{i}")?,
100        Variant::Int64(i) => write!(json_buffer, "{i}")?,
101        Variant::Float(f) => write!(json_buffer, "{f}")?,
102        Variant::Double(f) => write!(json_buffer, "{f}")?,
103        Variant::Decimal4(decimal) => write!(json_buffer, "{decimal}")?,
104        Variant::Decimal8(decimal) => write!(json_buffer, "{decimal}")?,
105        Variant::Decimal16(decimal) => write!(json_buffer, "{decimal}")?,
106        Variant::Date(date) => write!(json_buffer, "\"{}\"", format_date_string(date))?,
107        Variant::TimestampMicros(ts) => write!(json_buffer, "\"{}\"", ts.to_rfc3339())?,
108        Variant::TimestampNtzMicros(ts) => {
109            write!(json_buffer, "\"{}\"", format_timestamp_ntz_string(ts))?
110        }
111        Variant::Binary(bytes) => {
112            // Encode binary as base64 string
113            let base64_str = format_binary_base64(bytes);
114            let json_str = serde_json::to_string(&base64_str).map_err(|e| {
115                ArrowError::InvalidArgumentError(format!("JSON encoding error: {e}"))
116            })?;
117            write!(json_buffer, "{json_str}")?
118        }
119        Variant::String(s) => {
120            // Use serde_json to properly escape the string
121            let json_str = serde_json::to_string(s).map_err(|e| {
122                ArrowError::InvalidArgumentError(format!("JSON encoding error: {e}"))
123            })?;
124            write!(json_buffer, "{json_str}")?
125        }
126        Variant::ShortString(s) => {
127            // Use serde_json to properly escape the string
128            let json_str = serde_json::to_string(s.as_str()).map_err(|e| {
129                ArrowError::InvalidArgumentError(format!("JSON encoding error: {e}"))
130            })?;
131            write!(json_buffer, "{json_str}")?
132        }
133        Variant::Object(obj) => {
134            convert_object_to_json(json_buffer, obj)?;
135        }
136        Variant::List(arr) => {
137            convert_array_to_json(json_buffer, arr)?;
138        }
139    }
140    Ok(())
141}

143/// Convert object fields to JSON
144fn convert_object_to_json(buffer: &mut impl Write, obj: &VariantObject) -> Result<(), ArrowError> {
145    write!(buffer, "{{")?;
146
147    // Get all fields from the object
148    let mut first = true;
149
150    for (key, value) in obj.iter() {
151        if !first {
152            write!(buffer, ",")?;
153        }
154        first = false;
155
156        // Write the key (properly escaped)
157        let json_key = serde_json::to_string(key).map_err(|e| {
158            ArrowError::InvalidArgumentError(format!("JSON key encoding error: {e}"))
159        })?;
160        write!(buffer, "{json_key}:")?;
161
162        // Recursively convert the value
163        variant_to_json(buffer, &value)?;
164    }
165
166    write!(buffer, "}}")?;
167    Ok(())
168}

170/// Convert array elements to JSON
171fn convert_array_to_json(buffer: &mut impl Write, arr: &VariantList) -> Result<(), ArrowError> {
172    write!(buffer, "[")?;
173
174    let mut first = true;
175    for element in arr.iter() {
176        if !first {
177            write!(buffer, ",")?;
178        }
179        first = false;
180
181        variant_to_json(buffer, &element)?;
182    }
183
184    write!(buffer, "]")?;
185    Ok(())
186}

188/// Convert [`Variant`] to JSON [`String`]
189///
190/// This is a convenience function that converts a Variant to a JSON string.
191/// This is the same as calling [`variant_to_json`] with a [`Vec`].
192/// It's the simplest way to get a JSON representation when you just need a String result.
193///
194/// # Arguments
195///
196/// * `variant` - The Variant value to convert
197///
198/// # Returns
199///
200/// * `Ok(String)` containing the JSON representation
201/// * `Err` with error details if conversion fails
202///
203/// # Examples
204///
205/// ```rust
206/// # use parquet_variant::{Variant, variant_to_json_string};
207/// # use arrow_schema::ArrowError;
208/// let variant = Variant::Int32(42);
209/// let json = variant_to_json_string(&variant)?;
210/// assert_eq!(json, "42");
211/// # Ok::<(), ArrowError>(())
212/// ```
213///
214/// # Example: Create a [`Variant::Object`] and convert to JSON
215///
216/// This example shows how to create an object with two fields and convert it to JSON:
217/// ```json
218/// {
219///   "first_name": "Jiaying",
220///   "last_name": "Li"
221/// }
222/// ```
223///
224/// ```rust
225/// # use parquet_variant::{Variant, VariantBuilder, variant_to_json_string};
226/// # use arrow_schema::ArrowError;
227/// let mut builder = VariantBuilder::new();
228/// // Create an object builder that will write fields to the object
229/// let mut object_builder = builder.new_object();
230/// object_builder.insert("first_name", "Jiaying");
231/// object_builder.insert("last_name", "Li");
232/// object_builder.finish();
233/// // Finish the builder to get the metadata and value
234/// let (metadata, value) = builder.finish();
235/// // Create the Variant and convert to JSON
236/// let variant = Variant::try_new(&metadata, &value)?;
237/// let json = variant_to_json_string(&variant)?;
238/// assert_eq!(r#"{"first_name":"Jiaying","last_name":"Li"}"#, json);
239/// # Ok::<(), ArrowError>(())
240/// ```
241pub fn variant_to_json_string(variant: &Variant) -> Result<String, ArrowError> {
242    let mut buffer = Vec::new();
243    variant_to_json(&mut buffer, variant)?;
244    String::from_utf8(buffer)
245        .map_err(|e| ArrowError::InvalidArgumentError(format!("UTF-8 conversion error: {e}")))
246}

248/// Convert [`Variant`] to [`serde_json::Value`]
249///
250/// This function converts a Variant to a [`serde_json::Value`], which is useful
251/// when you need to work with the JSON data programmatically or integrate with
252/// other serde-based JSON processing.
253///
254/// # Arguments
255///
256/// * `variant` - The Variant value to convert
257///
258/// # Returns
259///
260/// * `Ok(Value)` containing the JSON value
261/// * `Err` with error details if conversion fails
262///
263/// # Examples
264///
265/// ```rust
266/// # use parquet_variant::{Variant, variant_to_json_value};
267/// # use serde_json::Value;
268/// # use arrow_schema::ArrowError;
269/// let variant = Variant::from("hello");
270/// let json_value = variant_to_json_value(&variant)?;
271/// assert_eq!(json_value, Value::String("hello".to_string()));
272/// # Ok::<(), ArrowError>(())
273/// ```
274pub fn variant_to_json_value(variant: &Variant) -> Result<Value, ArrowError> {
275    match variant {
276        Variant::Null => Ok(Value::Null),
277        Variant::BooleanTrue => Ok(Value::Bool(true)),
278        Variant::BooleanFalse => Ok(Value::Bool(false)),
279        Variant::Int8(i) => Ok(Value::Number((*i).into())),
280        Variant::Int16(i) => Ok(Value::Number((*i).into())),
281        Variant::Int32(i) => Ok(Value::Number((*i).into())),
282        Variant::Int64(i) => Ok(Value::Number((*i).into())),
283        Variant::Float(f) => serde_json::Number::from_f64((*f).into())
284            .map(Value::Number)
285            .ok_or_else(|| ArrowError::InvalidArgumentError("Invalid float value".to_string())),
286        Variant::Double(f) => serde_json::Number::from_f64(*f)
287            .map(Value::Number)
288            .ok_or_else(|| ArrowError::InvalidArgumentError("Invalid double value".to_string())),
289        Variant::Decimal4(decimal4) => {
290            let scale = decimal4.scale();
291            let integer = decimal4.integer();
292
293            let integer = if scale == 0 {
294                integer
295            } else {
296                let divisor = 10_i32.pow(scale as u32);
297                if integer % divisor != 0 {
298                    // fall back to floating point
299                    return Ok(Value::from(integer as f64 / divisor as f64));
300                }
301                integer / divisor
302            };
303            Ok(Value::from(integer))
304        }
305        Variant::Decimal8(decimal8) => {
306            let scale = decimal8.scale();
307            let integer = decimal8.integer();
308
309            let integer = if scale == 0 {
310                integer
311            } else {
312                let divisor = 10_i64.pow(scale as u32);
313                if integer % divisor != 0 {
314                    // fall back to floating point
315                    return Ok(Value::from(integer as f64 / divisor as f64));
316                }
317                integer / divisor
318            };
319            Ok(Value::from(integer))
320        }
321        Variant::Decimal16(decimal16) => {
322            let scale = decimal16.scale();
323            let integer = decimal16.integer();
324
325            let integer = if scale == 0 {
326                integer
327            } else {
328                let divisor = 10_i128.pow(scale as u32);
329                if integer % divisor != 0 {
330                    // fall back to floating point
331                    return Ok(Value::from(integer as f64 / divisor as f64));
332                }
333                integer / divisor
334            };
335            // i128 has higher precision than any 64-bit type. Try a lossless narrowing cast to
336            // i64 or u64 first, falling back to a lossy narrowing cast to f64 if necessary.
337            let value = i64::try_from(integer)
338                .map(Value::from)
339                .or_else(|_| u64::try_from(integer).map(Value::from))
340                .unwrap_or_else(|_| Value::from(integer as f64));
341            Ok(value)
342        }
343        Variant::Date(date) => Ok(Value::String(format_date_string(date))),
344        Variant::TimestampMicros(ts) => Ok(Value::String(ts.to_rfc3339())),
345        Variant::TimestampNtzMicros(ts) => Ok(Value::String(format_timestamp_ntz_string(ts))),
346        Variant::Binary(bytes) => Ok(Value::String(format_binary_base64(bytes))),
347        Variant::String(s) => Ok(Value::String(s.to_string())),
348        Variant::ShortString(s) => Ok(Value::String(s.to_string())),
349        Variant::Object(obj) => {
350            let map = obj
351                .iter()
352                .map(|(k, v)| variant_to_json_value(&v).map(|json_val| (k.to_string(), json_val)))
353                .collect::<Result<_, _>>()?;
354            Ok(Value::Object(map))
355        }
356        Variant::List(arr) => {
357            let vec = arr
358                .iter()
359                .map(|element| variant_to_json_value(&element))
360                .collect::<Result<_, _>>()?;
361            Ok(Value::Array(vec))
362        }
363    }
364}

530    /// Reusable test structure for JSON conversion testing
531    struct JsonTest {
532        variant: Variant<'static, 'static>,
533        expected_json: &'static str,
534        expected_value: Value,
535    }
536
537    impl JsonTest {
538        fn run(self) {
539            let json_string = variant_to_json_string(&self.variant)
540                .expect("variant_to_json_string should succeed");
541            assert_eq!(
542                json_string, self.expected_json,
543                "JSON string mismatch for variant: {:?}",
544                self.variant
545            );
546
547            let json_value =
548                variant_to_json_value(&self.variant).expect("variant_to_json_value should succeed");
549
550            // For floating point numbers, we need special comparison due to JSON number representation
551            match (&json_value, &self.expected_value) {
552                (Value::Number(actual), Value::Number(expected)) => {
553                    let actual_f64 = actual.as_f64().unwrap_or(0.0);
554                    let expected_f64 = expected.as_f64().unwrap_or(0.0);
555                    assert!(
556                        (actual_f64 - expected_f64).abs() < f64::EPSILON,
557                        "JSON value mismatch for variant: {:?}, got {}, expected {}",
558                        self.variant,
559                        actual_f64,
560                        expected_f64
561                    );
562                }
563                _ => {
564                    assert_eq!(
565                        json_value, self.expected_value,
566                        "JSON value mismatch for variant: {:?}",
567                        self.variant
568                    );
569                }
570            }
571
572            // Verify roundtrip: JSON string should parse to same value
573            let parsed: Value =
574                serde_json::from_str(&json_string).expect("Generated JSON should be valid");
575            // Same floating point handling for roundtrip
576            match (&parsed, &self.expected_value) {
577                (Value::Number(actual), Value::Number(expected)) => {
578                    let actual_f64 = actual.as_f64().unwrap_or(0.0);
579                    let expected_f64 = expected.as_f64().unwrap_or(0.0);
580                    assert!(
581                        (actual_f64 - expected_f64).abs() < f64::EPSILON,
582                        "Parsed JSON mismatch for variant: {:?}, got {}, expected {}",
583                        self.variant,
584                        actual_f64,
585                        expected_f64
586                    );
587                }
588                _ => {
589                    assert_eq!(
590                        parsed, self.expected_value,
591                        "Parsed JSON mismatch for variant: {:?}",
592                        self.variant
593                    );
594                }
595            }
596        }
597    }
