34/// Concatenate multiple [`RecordBatch`]es
35///
36/// Implements the common pattern of incrementally creating output
37/// [`RecordBatch`]es of a specific size from an input stream of
38/// [`RecordBatch`]es.
39///
40/// This is useful after operations such as [`filter`] and [`take`] that produce
41/// smaller batches, and we want to coalesce them into larger
42///
43/// [`filter`]: crate::filter::filter
44/// [`take`]: crate::take::take
45///
46/// See: <https://github.com/apache/arrow-rs/issues/6692>
47///
48/// # Example
49/// ```
50/// use arrow_array::record_batch;
51/// use arrow_select::coalesce::{BatchCoalescer};
52/// let batch1 = record_batch!(("a", Int32, [1, 2, 3])).unwrap();
53/// let batch2 = record_batch!(("a", Int32, [4, 5])).unwrap();
54///
55/// // Create a `BatchCoalescer` that will produce batches with at least 4 rows
56/// let target_batch_size = 4;
57/// let mut coalescer = BatchCoalescer::new(batch1.schema(), 4);
58///
59/// // push the batches
60/// coalescer.push_batch(batch1).unwrap();
61/// // only pushed 3 rows (not yet 4, enough to produce a batch)
62/// assert!(coalescer.next_completed_batch().is_none());
63/// coalescer.push_batch(batch2).unwrap();
64/// // now we have 5 rows, so we can produce a batch
65/// let finished = coalescer.next_completed_batch().unwrap();
66/// // 4 rows came out (target batch size is 4)
67/// let expected = record_batch!(("a", Int32, [1, 2, 3, 4])).unwrap();
68/// assert_eq!(finished, expected);
69///
70/// // Have no more input, but still have an in-progress batch
71/// assert!(coalescer.next_completed_batch().is_none());
72/// // We can finish the batch, which will produce the remaining rows
73/// coalescer.finish_buffered_batch().unwrap();
74/// let expected = record_batch!(("a", Int32, [5])).unwrap();
75/// assert_eq!(coalescer.next_completed_batch().unwrap(), expected);
76///
77/// // The coalescer is now empty
78/// assert!(coalescer.next_completed_batch().is_none());
79/// ```
80///
81/// # Background
82///
83/// Generally speaking, larger [`RecordBatch`]es are more efficient to process
84/// than smaller [`RecordBatch`]es (until the CPU cache is exceeded) because
85/// there is fixed processing overhead per batch. This coalescer builds up these
86/// larger batches incrementally.
87///
88/// ```text
89/// ┌────────────────────┐
90/// │    RecordBatch     │
91/// │   num_rows = 100   │
92/// └────────────────────┘                 ┌────────────────────┐
93///                                        │                    │
94/// ┌────────────────────┐     Coalesce    │                    │
95/// │                    │      Batches    │                    │
96/// │    RecordBatch     │                 │                    │
97/// │   num_rows = 200   │  ─ ─ ─ ─ ─ ─ ▶  │                    │
98/// │                    │                 │    RecordBatch     │
99/// │                    │                 │   num_rows = 400   │
100/// └────────────────────┘                 │                    │
101///                                        │                    │
102/// ┌────────────────────┐                 │                    │
103/// │                    │                 │                    │
104/// │    RecordBatch     │                 │                    │
105/// │   num_rows = 100   │                 └────────────────────┘
106/// │                    │
107/// └────────────────────┘
108/// ```
109///
110/// # Notes:
111///
112/// 1. Output rows are produced in the same order as the input rows
113///
114/// 2. The output is a sequence of batches, with all but the last being at exactly
115///    `target_batch_size` rows.
116///
117/// 3. Eventually this may also be able to handle other optimizations such as a
118///    combined filter/coalesce operation. See <https://github.com/apache/arrow-rs/issues/6692>
119///
120#[derive(Debug)]
121pub struct BatchCoalescer {
122    /// The input schema
123    schema: SchemaRef,
124    /// output batch size
125    batch_size: usize,
126    /// In-progress buffered batches
127    buffer: Vec<RecordBatch>,
128    /// Buffered row count. Always less than `batch_size`
129    buffered_rows: usize,
130    /// Completed batches
131    completed: VecDeque<RecordBatch>,
132}
133
134impl BatchCoalescer {
135    /// Create a new `BatchCoalescer`
136    ///
137    /// # Arguments
138    /// - `schema` - the schema of the output batches
139    /// - `batch_size` - the number of rows in each output batch.
140    ///   Typical values are `4096` or `8192` rows.
141    ///
142    pub fn new(schema: SchemaRef, batch_size: usize) -> Self {
143        Self {
144            schema,
145            batch_size,
146            buffer: vec![],
147            // We will for sure store at least one completed batch
148            completed: VecDeque::with_capacity(1),
149            buffered_rows: 0,
150        }
151    }
152
153    /// Return the schema of the output batches
154    pub fn schema(&self) -> SchemaRef {
155        Arc::clone(&self.schema)
156    }
157
158    /// Push next batch into the Coalescer
159    ///
160    /// See [`Self::next_completed_batch()`] to retrieve any completed batches.
161    pub fn push_batch(&mut self, batch: RecordBatch) -> Result<(), ArrowError> {
162        if batch.num_rows() == 0 {
163            // If the batch is empty, we don't need to do anything
164            return Ok(());
165        }
166
167        let mut batch = gc_string_view_batch(&batch);
168
169        // If pushing this batch would exceed the target batch size,
170        // finish the current batch and start a new one
171        while batch.num_rows() > (self.batch_size - self.buffered_rows) {
172            let remaining_rows = self.batch_size - self.buffered_rows;
173            debug_assert!(remaining_rows > 0);
174            let head_batch = batch.slice(0, remaining_rows);
175            batch = batch.slice(remaining_rows, batch.num_rows() - remaining_rows);
176            self.buffered_rows += head_batch.num_rows();
177            self.buffer.push(head_batch);
178            self.finish_buffered_batch()?;
179        }
180        // Add the remaining rows to the buffer
181        self.buffered_rows += batch.num_rows();
182        self.buffer.push(batch);
183
184        // If we have reached the target batch size, finalize the buffered batch
185        if self.buffered_rows >= self.batch_size {
186            self.finish_buffered_batch()?;
187        }
188        Ok(())
189    }
190
191    /// Concatenates any buffered batches into a single `RecordBatch` and
192    /// clears any output buffers
193    ///
194    /// Normally this is called when the input stream is exhausted, and
195    /// we want to finalize the last batch of rows.
196    ///
197    /// See [`Self::next_completed_batch()`] for the completed batches.
198    pub fn finish_buffered_batch(&mut self) -> Result<(), ArrowError> {
199        if self.buffer.is_empty() {
200            return Ok(());
201        }
202        let batch = concat_batches(&self.schema, &self.buffer)?;
203        self.buffer.clear();
204        self.buffered_rows = 0;
205        self.completed.push_back(batch);
206        Ok(())
207    }
208
209    /// Returns true if there is any buffered data
210    pub fn is_empty(&self) -> bool {
211        self.buffer.is_empty() && self.completed.is_empty()
212    }
213
214    /// Returns true if there are any completed batches
215    pub fn has_completed_batch(&self) -> bool {
216        !self.completed.is_empty()
217    }
218
219    /// Returns the next completed batch, if any
220    pub fn next_completed_batch(&mut self) -> Option<RecordBatch> {
221        self.completed.pop_front()
222    }
223}

225/// Heuristically compact `StringViewArray`s to reduce memory usage, if needed
226///
227/// Decides when to consolidate the StringView into a new buffer to reduce
228/// memory usage and improve string locality for better performance.
229///
230/// This differs from `StringViewArray::gc` because:
231/// 1. It may not compact the array depending on a heuristic.
232/// 2. It uses a precise block size to reduce the number of buffers to track.
233///
234/// # Heuristic
235///
236/// If the average size of each view is larger than 32 bytes, we compact the array.
237///
238/// `StringViewArray` include pointers to buffer that hold the underlying data.
239/// One of the great benefits of `StringViewArray` is that many operations
240/// (e.g., `filter`) can be done without copying the underlying data.
241///
242/// However, after a while (e.g., after `FilterExec` or `HashJoinExec`) the
243/// `StringViewArray` may only refer to a small portion of the buffer,
244/// significantly increasing memory usage.
245fn gc_string_view_batch(batch: &RecordBatch) -> RecordBatch {
246    let new_columns: Vec<ArrayRef> = batch
247        .columns()
248        .iter()
249        .map(|c| {
250            // Try to re-create the `StringViewArray` to prevent holding the underlying buffer too long.
251            let Some(s) = c.as_string_view_opt() else {
252                return Arc::clone(c);
253            };
254            let ideal_buffer_size: usize = s
255                .views()
256                .iter()
257                .map(|v| {
258                    let len = (*v as u32) as usize;
259                    if len > 12 {
260                        len
261                    } else {
262                        0
263                    }
264                })
265                .sum();
266            let actual_buffer_size = s.get_buffer_memory_size();
267
268            // Re-creating the array copies data and can be time consuming.
269            // We only do it if the array is sparse
270            if actual_buffer_size > (ideal_buffer_size * 2) {
271                // We set the block size to `ideal_buffer_size` so that the new StringViewArray only has one buffer, which accelerate later concat_batches.
272                // See https://github.com/apache/arrow-rs/issues/6094 for more details.
273                let mut builder = StringViewBuilder::with_capacity(s.len());
274                if ideal_buffer_size > 0 {
275                    builder = builder.with_fixed_block_size(ideal_buffer_size as u32);
276                }
277
278                for v in s.iter() {
279                    builder.append_option(v);
280                }
281
282                let gc_string = builder.finish();
283
284                debug_assert!(gc_string.data_buffers().len() <= 1); // buffer count can be 0 if the `ideal_buffer_size` is 0
285
286                Arc::new(gc_string)
287            } else {
288                Arc::clone(c)
289            }
290        })
291        .collect();
292    let mut options = RecordBatchOptions::new();
293    options = options.with_row_count(Some(batch.num_rows()));
294    RecordBatch::try_new_with_options(batch.schema(), new_columns, &options)
295        .expect("Failed to re-create the gc'ed record batch")
296}

384    /// Test for [`BatchCoalescer`]
385    ///
386    /// Pushes the input batches to the coalescer and verifies that the resulting
387    /// batches have the expected number of rows and contents.
388    #[derive(Debug, Clone, Default)]
389    struct Test {
390        /// Batches to feed to the coalescer. Tests must have at least one
391        /// schema
392        input_batches: Vec<RecordBatch>,
393
394        /// The schema. If not provided, the first batch's schema is used.
395        schema: Option<SchemaRef>,
396
397        /// Expected output sizes of the resulting batches
398        expected_output_sizes: Vec<usize>,
399        /// target batch size
400        target_batch_size: usize,
401    }
402
403    impl Test {
404        fn new() -> Self {
405            Self::default()
406        }
407
408        /// Set the target batch size
409        fn with_batch_size(mut self, target_batch_size: usize) -> Self {
410            self.target_batch_size = target_batch_size;
411            self
412        }
413
414        /// Extend the input batches with `batch`
415        fn with_batch(mut self, batch: RecordBatch) -> Self {
416            self.input_batches.push(batch);
417            self
418        }
419
420        /// Extends the input batches with `batches`
421        fn with_batches(mut self, batches: impl IntoIterator<Item = RecordBatch>) -> Self {
422            self.input_batches.extend(batches);
423            self
424        }
425
426        /// Specifies the schema for the test
427        fn with_schema(mut self, schema: SchemaRef) -> Self {
428            self.schema = Some(schema);
429            self
430        }
431
432        /// Extends `sizes` to expected output sizes
433        fn with_expected_output_sizes(mut self, sizes: impl IntoIterator<Item = usize>) -> Self {
434            self.expected_output_sizes.extend(sizes);
435            self
436        }
437
438        /// Runs the test -- see documentation on [`Test`] for details
439        fn run(self) {
440            let Self {
441                input_batches,
442                schema,
443                target_batch_size,
444                expected_output_sizes,
445            } = self;
446
447            let schema = schema.unwrap_or_else(|| input_batches[0].schema());
448
449            // create a single large input batch for output comparison
450            let single_input_batch = concat_batches(&schema, &input_batches).unwrap();
451
452            let mut coalescer = BatchCoalescer::new(Arc::clone(&schema), target_batch_size);
453
454            for batch in input_batches {
455                coalescer.push_batch(batch).unwrap();
456            }
457            coalescer.finish_buffered_batch().unwrap();
458            let mut output_batches = vec![];
459            while let Some(batch) = coalescer.next_completed_batch() {
460                output_batches.push(batch);
461            }
462
463            // make sure we got the expected number of output batches and content
464            let mut starting_idx = 0;
465            assert_eq!(expected_output_sizes.len(), output_batches.len());
466            let actual_output_sizes: Vec<usize> =
467                output_batches.iter().map(|b| b.num_rows()).collect();
468            assert_eq!(
469                expected_output_sizes, actual_output_sizes,
470                "Unexpected number of rows in output batches\n\
471                Expected\n{expected_output_sizes:#?}\nActual:{actual_output_sizes:#?}"
472            );
473            for (i, (expected_size, batch)) in
474                expected_output_sizes.iter().zip(output_batches).enumerate()
475            {
476                // compare the contents of the batch after normalization (using
477                // `==` compares the underlying memory layout too)
478                let expected_batch = single_input_batch.slice(starting_idx, *expected_size);
479                let expected_batch = normalize_batch(expected_batch);
480                let batch = normalize_batch(batch);
481                assert_eq!(
482                    expected_batch, batch,
483                    "Unexpected content in batch {i}:\
484                    \n\nExpected:\n{expected_batch:#?}\n\nActual:\n{batch:#?}"
485                );
486                starting_idx += *expected_size;
487            }
488        }
489    }

567    /// runs garbage collection on string view array
568    /// and ensures the number of rows are the same
569    fn do_gc(array: StringViewArray) -> StringViewArray {
570        let batch = RecordBatch::try_from_iter(vec![("a", Arc::new(array) as ArrayRef)]).unwrap();
571        let gc_batch = gc_string_view_batch(&batch);
572        assert_eq!(batch.num_rows(), gc_batch.num_rows());
573        assert_eq!(batch.schema(), gc_batch.schema());
574        gc_batch
575            .column(0)
576            .as_any()
577            .downcast_ref::<StringViewArray>()
578            .unwrap()
579            .clone()
580    }

582    /// Describes parameters for creating a `StringViewArray`
583    struct StringViewTest {
584        /// The number of rows in the array
585        rows: usize,
586        /// The strings to use in the array (repeated over and over
587        strings: Vec<Option<&'static str>>,
588    }

590    impl StringViewTest {
591        /// Create a `StringViewArray` with the parameters specified in this struct
592        fn build(self) -> StringViewArray {
593            let mut builder = StringViewBuilder::with_capacity(100).with_fixed_block_size(8192);
594            loop {
595                for &v in self.strings.iter() {
596                    builder.append_option(v);
597                    if builder.len() >= self.rows {
598                        return builder.finish();
599                    }
600                }
601            }
602        }
603    }

605    /// Normalize the `RecordBatch` so that the memory layout is consistent
606    /// (e.g. StringArray is compacted).
607    fn normalize_batch(batch: RecordBatch) -> RecordBatch {
608        // Only need to normalize StringViews (as == also tests for memory layout)
609        let (schema, mut columns, row_count) = batch.into_parts();
610
611        for column in columns.iter_mut() {
612            let Some(string_view) = column.as_string_view_opt() else {
613                continue;
614            };
615
616            // Re-create the StringViewArray to ensure memory layout is
617            // consistent
618            let mut builder = StringViewBuilder::new();
619            for s in string_view.iter() {
620                builder.append_option(s);
621            }
622            // Update the column with the new StringViewArray
623            *column = Arc::new(builder.finish());
624        }
625
626        let options = RecordBatchOptions::new().with_row_count(Some(row_count));
627        RecordBatch::try_new_with_options(schema, columns, &options).unwrap()
628    }
