18//! Module for transforming a typed arrow `Array` to `VariantArray`.
19
20use arrow::datatypes::{self, ArrowPrimitiveType, ArrowTimestampType, Date32Type};
21use parquet_variant::Variant;

23/// Options for controlling the behavior of `cast_to_variant_with_options`.
24#[derive(Debug, Clone, PartialEq, Eq)]
25pub struct CastOptions {
26    /// If true, return error on conversion failure. If false, insert null for failed conversions.
27    pub strict: bool,
28}
29
30impl Default for CastOptions {
31    fn default() -> Self {
32        Self { strict: true }
33    }
34}
35
36/// Extension trait for Arrow primitive types that can extract their native value from a Variant
37pub(crate) trait PrimitiveFromVariant: ArrowPrimitiveType {
38    fn from_variant(variant: &Variant<'_, '_>) -> Option<Self::Native>;
39}
40
41/// Extension trait for Arrow timestamp types that can extract their native value from a Variant
42/// We can't use [`PrimitiveFromVariant`] directly because we need _two_ implementations for each
43/// timestamp type -- the `NTZ` param here.
44pub(crate) trait TimestampFromVariant<const NTZ: bool>: ArrowTimestampType {
45    fn from_variant(variant: &Variant<'_, '_>) -> Option<Self::Native>;
46}
47
48/// Macro to generate PrimitiveFromVariant implementations for Arrow primitive types
49macro_rules! impl_primitive_from_variant {
50    ($arrow_type:ty, $variant_method:ident $(, $cast_fn:expr)?) => {
51        impl PrimitiveFromVariant for $arrow_type {
52            fn from_variant(variant: &Variant<'_, '_>) -> Option<Self::Native> {
53                let value = variant.$variant_method();
54                $( let value = value.map($cast_fn); )?
55                value
56            }
57        }
58    };
59}
60
61macro_rules! impl_timestamp_from_variant {
62    ($timestamp_type:ty, $variant_method:ident, ntz=$ntz:ident, $cast_fn:expr $(,)?) => {
63        impl TimestampFromVariant<{ $ntz }> for $timestamp_type {
64            fn from_variant(variant: &Variant<'_, '_>) -> Option<Self::Native> {
65                variant.$variant_method().and_then($cast_fn)
66            }
67        }
68    };
69}

112/// Convert the value at a specific index in the given array into a `Variant`.
113macro_rules! non_generic_conversion_single_value {
114    ($array:expr, $cast_fn:expr, $index:expr) => {{
115        let array = $array;
116        if array.is_null($index) {
117            Variant::Null
118        } else {
119            let cast_value = $cast_fn(array.value($index));
120            Variant::from(cast_value)
121        }
122    }};
123}
124pub(crate) use non_generic_conversion_single_value;

126/// Convert the value at a specific index in the given array into a `Variant`,
127/// using `method` requiring a generic type to downcast the generic array
128/// to a specific array type and `cast_fn` to transform the element.
129macro_rules! generic_conversion_single_value {
130    ($t:ty, $method:ident, $cast_fn:expr, $input:expr, $index:expr) => {{
131        $crate::type_conversion::non_generic_conversion_single_value!(
132            $input.$method::<$t>(),
133            $cast_fn,
134            $index
135        )
136    }};
137}
138pub(crate) use generic_conversion_single_value;

140/// Convert the value at a specific index in the given array into a `Variant`.
141macro_rules! primitive_conversion_single_value {
142    ($t:ty, $input:expr, $index:expr) => {{
143        $crate::type_conversion::generic_conversion_single_value!(
144            $t,
145            as_primitive,
146            |v| v,
147            $input,
148            $index
149        )
150    }};
151}
152pub(crate) use primitive_conversion_single_value;

154/// Convert a decimal value to a `VariantDecimal`
155macro_rules! decimal_to_variant_decimal {
156    ($v:ident, $scale:expr, $value_type:ty, $variant_type:ty) => {{
157        let (v, scale) = if *$scale < 0 {
158            // For negative scale, we need to multiply the value by 10^|scale|
159            // For example: 123 with scale -2 becomes 12300 with scale 0
160            let multiplier = <$value_type>::pow(10, (-*$scale) as u32);
161            (<$value_type>::checked_mul($v, multiplier), 0u8)
162        } else {
163            (Some($v), *$scale as u8)
164        };
165
166        // Return an Option to allow callers to decide whether to error (strict)
167        // or append null (non-strict) on conversion failure
168        v.and_then(|v| <$variant_type>::try_new(v, scale).ok())
169    }};
170}
171pub(crate) use decimal_to_variant_decimal;
