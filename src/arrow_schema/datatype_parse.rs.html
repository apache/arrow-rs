22/// Parses a DataType from a string representation
23///
24/// For example, the string "Int32" would be parsed into [`DataType::Int32`]
25pub(crate) fn parse_data_type(val: &str) -> ArrowResult<DataType> {
26    Parser::new(val).parse()
27}
28
29type ArrowResult<T> = Result<T, ArrowError>;
30
31fn make_error(val: &str, msg: &str) -> ArrowError {
32    let msg = format!(
33        "Unsupported type '{val}'. Must be a supported arrow type name such as 'Int32' or 'Timestamp(ns)'. Error {msg}"
34    );
35    ArrowError::ParseError(msg)
36}
37
38fn make_error_expected(val: &str, expected: &Token, actual: &Token) -> ArrowError {
39    make_error(val, &format!("Expected '{expected}', got '{actual}'"))
40}

42/// Implementation of `parse_data_type`, modeled after <https://github.com/sqlparser-rs/sqlparser-rs>
43#[derive(Debug)]
44struct Parser<'a> {
45    val: &'a str,
46    tokenizer: Peekable<Tokenizer<'a>>,
47}
48
49impl<'a> Parser<'a> {
50    fn new(val: &'a str) -> Self {
51        Self {
52            val,
53            tokenizer: Tokenizer::new(val).peekable(),
54        }
55    }
56
57    fn parse(mut self) -> ArrowResult<DataType> {
58        let data_type = self.parse_next_type()?;
59        // ensure that there is no trailing content
60        if self.tokenizer.next().is_some() {
61            Err(make_error(
62                self.val,
63                &format!("checking trailing content after parsing '{data_type}'"),
64            ))
65        } else {
66            Ok(data_type)
67        }
68    }
69
70    /// parses the next full DataType
71    fn parse_next_type(&mut self) -> ArrowResult<DataType> {
72        match self.next_token()? {
73            Token::SimpleType(data_type) => Ok(data_type),
74            Token::Timestamp => self.parse_timestamp(),
75            Token::Time32 => self.parse_time32(),
76            Token::Time64 => self.parse_time64(),
77            Token::Duration => self.parse_duration(),
78            Token::Interval => self.parse_interval(),
79            Token::FixedSizeBinary => self.parse_fixed_size_binary(),
80            Token::Decimal32 => self.parse_decimal_32(),
81            Token::Decimal64 => self.parse_decimal_64(),
82            Token::Decimal128 => self.parse_decimal_128(),
83            Token::Decimal256 => self.parse_decimal_256(),
84            Token::Dictionary => self.parse_dictionary(),
85            Token::List => self.parse_list(),
86            Token::ListView => self.parse_list_view(),
87            Token::LargeList => self.parse_large_list(),
88            Token::LargeListView => self.parse_large_list_view(),
89            Token::FixedSizeList => self.parse_fixed_size_list(),
90            Token::Struct => self.parse_struct(),
91            tok => Err(make_error(
92                self.val,
93                &format!("finding next type, got unexpected '{tok}'"),
94            )),
95        }
96    }

98    /// Parses list field name. Returns default field name if not found.
99    fn parse_list_field_name(&mut self, context: &str) -> ArrowResult<String> {
100        // field must be after a comma
101        if self
102            .tokenizer
103            .next_if(|next| matches!(next, Ok(Token::Comma)))
104            .is_none()
105        {
106            return Ok(Field::LIST_FIELD_DEFAULT_NAME.into());
107        }
108
109        // expects: `field: 'field_name'`.
110        self.expect_token(Token::Field)?;
111        self.expect_token(Token::Colon)?;
112        self.parse_single_quoted_string(context)
113    }

115    /// Parses the List type (called after `List` has been consumed)
116    /// E.g: List(nullable Int64, field: 'foo')
117    fn parse_list(&mut self) -> ArrowResult<DataType> {
118        self.expect_token(Token::LParen)?;
119        let nullable = self.parse_opt_nullable();
120        let data_type = self.parse_next_type()?;
121        let field = self.parse_list_field_name("List")?;
122        self.expect_token(Token::RParen)?;
123        Ok(DataType::List(Arc::new(Field::new(
124            field, data_type, nullable,
125        ))))

264    /// Parses the next timestamp (called after `Timestamp` has been consumed)
265    fn parse_timestamp(&mut self) -> ArrowResult<DataType> {
266        self.expect_token(Token::LParen)?;
267        let time_unit = self.parse_time_unit("Timestamp")?;
268
269        let timezone;
270        match self.next_token()? {
271            Token::Comma => {
272                match self.next_token()? {
273                    // Support old style `Timestamp(Nanosecond, None)`
274                    Token::None => {
275                        timezone = None;
276                    }
277                    // Support old style `Timestamp(Nanosecond, Some("Timezone"))`
278                    Token::Some => {
279                        self.expect_token(Token::LParen)?;
280                        timezone = Some(self.parse_double_quoted_string("Timezone")?);
281                        self.expect_token(Token::RParen)?;
282                    }
283                    Token::DoubleQuotedString(tz) => {
284                        // Support new style `Timestamp(Nanosecond, "Timezone")`
285                        timezone = Some(tz);
286                    }
287                    tok => {
288                        return Err(make_error(
289                            self.val,
290                            &format!("Expected None, Some, or a timezone string, got {tok:?}"),
291                        ));
292                    }
293                };
294                self.expect_token(Token::RParen)?;
295            }
296            // No timezone (e.g `Timestamp(ns)`)
297            Token::RParen => {
298                timezone = None;
299            }
300            next_token => {
301                return Err(make_error(
302                    self.val,
303                    &format!("Expected comma followed by a timezone, or an ), got {next_token:?}"),
304                ));
305            }
306        }
307        Ok(DataType::Timestamp(time_unit, timezone.map(Into::into)))
308    }

399    fn parse_dictionary(&mut self) -> ArrowResult<DataType> {
400        self.expect_token(Token::LParen)?;
401        let key_type = self.parse_next_type()?;
402        self.expect_token(Token::Comma)?;
403        let value_type = self.parse_next_type()?;
404        self.expect_token(Token::RParen)?;

406            Box::new(key_type),
407            Box::new(value_type),

411    /// Parses the next Struct (called after `Struct` has been consumed)
412    fn parse_struct(&mut self) -> ArrowResult<DataType> {
413        self.expect_token(Token::LParen)?;
414        let mut fields = Vec::new();
415        loop {
416            // expects:   "field name": [nullable] #datatype
417
418            let field_name = match self.next_token()? {
419                Token::RParen => {
420                    break;
421                }
422                Token::DoubleQuotedString(field_name) => field_name,
423                tok => {
424                    return Err(make_error(
425                        self.val,
426                        &format!("Expected a double quoted string for a field name; got {tok:?}"),
427                    ));
428                }
429            };
430            self.expect_token(Token::Colon)?;
431
432            let nullable = self.parse_opt_nullable();
433            let field_type = self.parse_next_type()?;
434            fields.push(Arc::new(Field::new(field_name, field_type, nullable)));
435            match self.next_token()? {
436                Token::Comma => continue,
437                Token::RParen => break,
438                tok => {
439                    return Err(make_error(
440                        self.val,
441                        &format!(
442                            "Unexpected token while parsing Struct fields. Expected ',' or ')', but got '{tok}'"
443                        ),
444                    ));
445                }
446            }
447        }
448        Ok(DataType::Struct(Fields::from(fields)))
449    }
450
451    /// return and consume if the next token is `Token::Nullable`
452    fn parse_opt_nullable(&mut self) -> bool {
453        self.tokenizer
454            .next_if(|next| matches!(next, Ok(Token::Nullable)))
455            .is_some()
456    }
457
458    /// return the next token, or an error if there are none left
459    fn next_token(&mut self) -> ArrowResult<Token> {
460        match self.tokenizer.next() {
461            None => Err(make_error(self.val, "finding next token")),
462            Some(token) => token,
463        }
464    }
465
466    /// consume the next token, returning OK(()) if it matches tok, and Err if not
467    fn expect_token(&mut self, tok: Token) -> ArrowResult<()> {
468        let next_token = self.next_token()?;
469        if next_token == tok {
470            Ok(())
471        } else {
472            Err(make_error_expected(self.val, &tok, &next_token))
473        }
474    }
475}

477/// returns true if this character is a separator
478fn is_separator(c: char) -> bool {
479    c == '(' || c == ')' || c == ',' || c == ':' || c == ' '
480}

482enum QuoteType {
483    Double,
484    Single,
485}

487#[derive(Debug)]
488/// Splits a strings like Dictionary(Int32, Int64) into tokens sutable for parsing
489///
490/// For example the string "Timestamp(ns)" would be parsed into:
491///
492/// * Token::Timestamp
493/// * Token::Lparen
494/// * Token::IntervalUnit(IntervalUnit::Nanosecond)
495/// * Token::Rparen,
496struct Tokenizer<'a> {
497    val: &'a str,
498    chars: Peekable<Chars<'a>>,
499    // temporary buffer for parsing words
500    word: String,
501}
502
503impl<'a> Tokenizer<'a> {
504    fn new(val: &'a str) -> Self {
505        Self {
506            val,
507            chars: val.chars().peekable(),
508            word: String::new(),
509        }
510    }
511
512    /// returns the next char, without consuming it
513    fn peek_next_char(&mut self) -> Option<char> {
514        self.chars.peek().copied()
515    }
516
517    /// returns the next char, and consuming it
518    fn next_char(&mut self) -> Option<char> {
519        self.chars.next()
520    }
521
522    /// parse the characters in val starting at pos, until the next
523    /// `,`, `(`, or `)` or end of line
524    fn parse_word(&mut self) -> ArrowResult<Token> {
525        // reset temp space
526        self.word.clear();
527        loop {
528            match self.peek_next_char() {
529                None => break,
530                Some(c) if is_separator(c) => break,
531                Some(c) => {
532                    self.next_char();
533                    self.word.push(c);
534                }
535            }
536        }
537
538        if let Some(c) = self.word.chars().next() {
539            // if it started with a number, try parsing it as an integer
540            if c == '-' || c.is_numeric() {
541                let val: i64 = self.word.parse().map_err(|e| {
542                    make_error(self.val, &format!("parsing {} as integer: {e}", self.word))
543                })?;
544                return Ok(Token::Integer(val));
545            }
546        }
547
548        // figure out what the word was
549        let token = match self.word.as_str() {
550            "Null" => Token::SimpleType(DataType::Null),
551            "Boolean" => Token::SimpleType(DataType::Boolean),
552
553            "Int8" => Token::SimpleType(DataType::Int8),
554            "Int16" => Token::SimpleType(DataType::Int16),
555            "Int32" => Token::SimpleType(DataType::Int32),
556            "Int64" => Token::SimpleType(DataType::Int64),
557
558            "UInt8" => Token::SimpleType(DataType::UInt8),
559            "UInt16" => Token::SimpleType(DataType::UInt16),
560            "UInt32" => Token::SimpleType(DataType::UInt32),
561            "UInt64" => Token::SimpleType(DataType::UInt64),
562
563            "Utf8" => Token::SimpleType(DataType::Utf8),
564            "LargeUtf8" => Token::SimpleType(DataType::LargeUtf8),
565            "Utf8View" => Token::SimpleType(DataType::Utf8View),
566            "Binary" => Token::SimpleType(DataType::Binary),
567            "BinaryView" => Token::SimpleType(DataType::BinaryView),
568            "LargeBinary" => Token::SimpleType(DataType::LargeBinary),
569
570            "Float16" => Token::SimpleType(DataType::Float16),
571            "Float32" => Token::SimpleType(DataType::Float32),
572            "Float64" => Token::SimpleType(DataType::Float64),
573
574            "Date32" => Token::SimpleType(DataType::Date32),
575            "Date64" => Token::SimpleType(DataType::Date64),
576
577            "List" => Token::List,
578            "ListView" => Token::ListView,
579            "LargeList" => Token::LargeList,
580            "LargeListView" => Token::LargeListView,
581            "FixedSizeList" => Token::FixedSizeList,
582
583            "s" | "Second" => Token::TimeUnit(TimeUnit::Second),
584            "ms" | "Millisecond" => Token::TimeUnit(TimeUnit::Millisecond),
585            "µs" | "us" | "Microsecond" => Token::TimeUnit(TimeUnit::Microsecond),
586            "ns" | "Nanosecond" => Token::TimeUnit(TimeUnit::Nanosecond),
587
588            "Timestamp" => Token::Timestamp,
589            "Time32" => Token::Time32,
590            "Time64" => Token::Time64,
591            "Duration" => Token::Duration,
592            "Interval" => Token::Interval,
593            "Dictionary" => Token::Dictionary,
594
595            "FixedSizeBinary" => Token::FixedSizeBinary,
596
597            "Decimal32" => Token::Decimal32,
598            "Decimal64" => Token::Decimal64,
599            "Decimal128" => Token::Decimal128,
600            "Decimal256" => Token::Decimal256,
601
602            "YearMonth" => Token::IntervalUnit(IntervalUnit::YearMonth),
603            "DayTime" => Token::IntervalUnit(IntervalUnit::DayTime),
604            "MonthDayNano" => Token::IntervalUnit(IntervalUnit::MonthDayNano),
605
606            "Some" => Token::Some,
607            "None" => Token::None,
608
609            "nullable" => Token::Nullable,
610            "field" => Token::Field,
611            "x" => Token::X,
612
613            "Struct" => Token::Struct,
614
615            token => {
616                return Err(make_error(self.val, &format!("unknown token: {token}")));
617            }
618        };
619        Ok(token)
620    }

622    /// Parses e.g. `"foo bar"`, `'foo bar'`
623    fn parse_quoted_string(&mut self, quote_type: QuoteType) -> ArrowResult<Token> {
624        let quote = match quote_type {
625            QuoteType::Double => '\"',
626            QuoteType::Single => '\'',
627        };
628
629        if self.next_char() != Some(quote) {
630            return Err(make_error(self.val, "Expected \""));
631        }
632
633        // reset temp space
634        self.word.clear();
635
636        let mut is_escaped = false;
637
638        loop {
639            match self.next_char() {
640                None => {
641                    return Err(ArrowError::ParseError(format!(
642                        "Unterminated string at: \"{}",
643                        self.word
644                    )));
645                }
646                Some(c) => match c {
647                    '\\' => {
648                        is_escaped = true;
649                        self.word.push(c);
650                    }
651                    c if c == quote => {
652                        if is_escaped {
653                            self.word.push(c);
654                            is_escaped = false;
655                        } else {
656                            break;
657                        }
658                    }
659                    c => {
660                        self.word.push(c);
661                    }
662                },
663            }
664        }
665
666        let val: String = self.word.parse().map_err(|err| {
667            ArrowError::ParseError(format!("Failed to parse string: \"{}\": {err}", self.word))
668        })?;
669
670        if val.is_empty() {
671            // Using empty strings as field names is just asking for trouble
672            return Err(make_error(self.val, "empty strings aren't allowed"));
673        }
674
675        match quote_type {
676            QuoteType::Double => Ok(Token::DoubleQuotedString(val)),
677            QuoteType::Single => Ok(Token::SingleQuotedString(val)),
678        }
679    }
680}
681
682impl Iterator for Tokenizer<'_> {
683    type Item = ArrowResult<Token>;
684
685    fn next(&mut self) -> Option<Self::Item> {
686        loop {
687            match self.peek_next_char()? {
688                ' ' => {
689                    // skip whitespace
690                    self.next_char();
691                    continue;
692                }
693                '"' => {
694                    return Some(self.parse_quoted_string(QuoteType::Double));
695                }
696                '\'' => {
697                    return Some(self.parse_quoted_string(QuoteType::Single));
698                }
699                '(' => {
700                    self.next_char();
701                    return Some(Ok(Token::LParen));
702                }
703                ')' => {
704                    self.next_char();
705                    return Some(Ok(Token::RParen));
706                }
707                ',' => {
708                    self.next_char();
709                    return Some(Ok(Token::Comma));
710                }
711                ':' => {
712                    self.next_char();
713                    return Some(Ok(Token::Colon));
714                }
715                _ => return Some(self.parse_word()),
716            }
717        }
718    }
719}

1257                "Error converting 500 into i8 for Decimal32: out of range integral type conversion attempted",
1258            ),

1278
1279        for (data_type_string, expected_message) in cases {

1282                Ok(d) => panic!("Expected error while parsing '{data_type_string}', but got '{d}'"),
1283                Err(e) => {

1286                        message.contains(expected_message),
1287                        "\n\ndid not find expected in actual.\n\nexpected: {expected_message}\nactual: {message}\n"
1288                    );
1289
1290                    if !message.contains("Unterminated string") {
1291                        // errors should also contain a help message
1292                        assert!(message.contains("Must be a supported arrow type name such as 'Int32' or 'Timestamp(ns)'"), "message: {message}");
1293                    }
1294                }
1295            }
1296        }
1297    }
